using System;
using System.Linq;
class Program
{
  static void Main() {
      
    int[] oddNum = new int[10];
    int[] evenNum = new int[10];
    int num = 0;
    
    for (int i=0; i<10; i++)
    {
        Console.WriteLine("Enter a number: ");
        num = Convert.ToInt32(Console.ReadLine());
        
        if (num % 2 == 0)
        {
            evenNum[i] = num;
        }
        else
        {
            oddNum[i] = num;
        }
    }
    
    Array.Sort(oddNum); // sort odd nums in asc
    Array.Sort(evenNum);
    Array.Reverse(evenNum); // sort evens then reverse to make desc
    
    /* I'm confused by the instructions here because it asks to sort the 
    arrays in step 6&7 but then wants the opposite order (asc/desc) in the
    foreach loops. So I just reversed again in the loops, hoping thats correct */
    
    Console.WriteLine("==Odd Numbers==");  
    foreach (int o in oddNum.Reverse())
    {
      Console.WriteLine(o);  
    }
    Console.WriteLine("==Even Numbers=="); 
    foreach (int e in evenNum.Reverse()) 
    {

      Console.WriteLine(e);  
    }
    
    Console.WriteLine("Sum of Odd Numbers = " + oddNum.Sum());
    Console.WriteLine("Sum of Even Numbers = " + evenNum.Sum());
  }
}
